set(TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET test_component_intfs_sched_serv_c_plugin)

set(BUILD_INTERFACE_INCLUDES
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_BINARY_DIR}/include
)

add_library(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} SHARED
    ${PROJECT_SOURCE_DIR}/src/fep3/components/scheduler/mock/mock_scheduler_service.h
    test_plugin/scheduler_service.cpp
    test_plugin/test_plugin.h
    test_plugin/test_plugin.cpp
)
# some include files are generated by fep3_participant so the plugin depends on fep3_participant
add_dependencies(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} fep3_participant)
target_include_directories(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} PUBLIC
    "$<BUILD_INTERFACE:${BUILD_INTERFACE_INCLUDES}>"
    "$<INSTALL_INTERFACE:${INSTALL_INTERFACE_INCLUDES}>"
)
target_link_libraries(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} PUBLIC a_util_result)
target_link_libraries(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} PRIVATE
    GTest::Main
    GMock::GMock
)
target_compile_definitions(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} PRIVATE _FEP3_PARTICIPANT_INCLUDED_STATIC)
set_target_properties(${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET} PROPERTIES FOLDER "test/private/component_intfs/scheduler/c/test_plugin")

##################################################################
#  test application
##################################################################

project(tester_component_intfs_sched_serv_c_plugin)

add_definitions(-DCURRENT_TEST_DIR="${CMAKE_CURRENT_SOURCE_DIR}/../")
add_definitions(-DTEST_PLUGIN_DIR="${CMAKE_CURRENT_BINARY_DIR}")

add_executable(${PROJECT_NAME}
    tester_sched_serv_intf_c_plugin.cpp
    )
target_include_directories(${PROJECT_NAME} PRIVATE
)
target_compile_definitions(${PROJECT_NAME} PRIVATE
    PLUGIN="$<TARGET_FILE:${TEST_COMPONENT_INTERFACES_SCHEDULER_SERVICE_C_PLUGIN_TARGET}>"
    TEST_BUILD_DIR="$<TARGET_FILE_DIR:${PROJECT_NAME}>"
)
target_link_libraries(${PROJECT_NAME} PRIVATE
    GTest::Main
    GMock::Main
    ${CMAKE_DL_LIBS}
    fep3_participant_private_lib
    participant_private_test_utils
)
target_link_libraries(${PROJECT_NAME} PUBLIC
    a_util
    a_util_result
)

add_test(NAME ${PROJECT_NAME}
    COMMAND ${PROJECT_NAME}
    TIMEOUT 10
    WORKING_DIRECTORY ".."
)
set_target_properties(${PROJECT_NAME} PROPERTIES FOLDER "test/private/component_intfs/scheduler/c")
